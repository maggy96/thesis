% Zentrierte, farbig hinterlegte Textbox
\newcommand\textbox[1]{%
  \vspace*{2mm}
  \begin{center}
    \fcolorbox{darkgray}{lightgray}
    {
      \parbox{0.80\textwidth}
      {
        \begin{flushleft}
          #1
        \end{flushleft}
      }
    }
  \end{center}
  \vspace*{2mm}
}
